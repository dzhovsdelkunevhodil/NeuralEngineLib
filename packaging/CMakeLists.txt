
set(CPACK_COMPONENTS_GROUPING ALL_COMPONENTS_SEPARATELY)

set(CPACK_PACKAGE_NAME "NeuralEngineLib")
set(CPACK_PACKAGE_VERSION ${PROJECT_VERSION})
set(CPACK_VERBATIM_VARIABLES TRUE)

if(WIN32)
    set(CPACK_GENERATOR "WIX")
    set(CPACK_PACKAGE_FILE_NAME "${CPACK_PACKAGE_NAME}-${PROJECT_VERSION}-win")
elseif(UNIX)
    set(CPACK_GENERATOR "DEB")
    set(CPACK_DEBIAN_PACKAGE_MAINTAINER "Matvey")
    set(CPACK_PACKAGE_FILE_NAME "${CPACK_PACKAGE_NAME}-${PROJECT_VERSION}-linux")
endif()


set(CPACK_COMPONENTS_ALL lib dev)
set(CPACK_COMPONENT_DEV_DESCRIPTION "Development files")
set(CPACK_COMPONENT_LIB_DESCRIPTION "Runtime libraries")
message(STATUS "Using CPACK_GENERATOR=${CPACK_GENERATOR}")
include(CPack)
